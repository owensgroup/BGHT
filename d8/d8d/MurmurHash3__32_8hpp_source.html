30 template <typename Key>
31 struct MurmurHash3_32 {
32  using key_type = Key;
33  using result_type = uint32_t;
34  __host__ __device__ constexpr MurmurHash3_32() : m_seed(0) {}
35 
36  __host__ __device__ constexpr MurmurHash3_32(uint32_t seed) : m_seed(seed) {}
37 
38  MurmurHash3_32(const MurmurHash3_32&) = default;
39  MurmurHash3_32(MurmurHash3_32&&) = default;
40  MurmurHash3_32& operator=(MurmurHash3_32 const&) = default;
41  MurmurHash3_32& operator=(MurmurHash3_32&&) = default;
42  ~MurmurHash3_32() = default;

44  constexpr uint32_t __host__ __device__ operator()(Key const& key) const noexcept {
45  constexpr int len = sizeof(Key);
46  const uint8_t* const data = (const uint8_t*)&key;
47  constexpr int nblocks = len / 4;
48 
49  uint32_t h1 = m_seed;
50  constexpr uint32_t c1 = 0xcc9e2d51;
51  constexpr uint32_t c2 = 0x1b873593;
52  //----------
53  // body
54  const uint32_t* const blocks = (const uint32_t*)(data + nblocks * 4);
55  for (int i = -nblocks; i; i++) {
56  uint32_t k1 = blocks[i]; // getblock32(blocks,i);
57  k1 *= c1;
58  k1 = rotl32(k1, 15);
59  k1 *= c2;
60  h1 ^= k1;
61  h1 = rotl32(h1, 13);
62  h1 = h1 * 5 + 0xe6546b64;
63  }
64  //----------
65  // tail
66  const uint8_t* tail = (const uint8_t*)(data + nblocks * 4);
67  uint32_t k1 = 0;
68  switch (len & 3) {
69  case 3:
70  k1 ^= tail[2] << 16;
71  case 2:
72  k1 ^= tail[1] << 8;
73  case 1:
74  k1 ^= tail[0];
75  k1 *= c1;
76  k1 = rotl32(k1, 15);
77  k1 *= c2;
78  h1 ^= k1;
79  };
80  //----------
81  // finalization
82  h1 ^= len;
83  h1 = fmix32(h1);
84  return h1;
85  }
86 
87  private:
88  constexpr __host__ __device__ uint32_t rotl32(uint32_t x, int8_t r) const noexcept {
89  return (x << r) | (x >> (32 - r));
90  }
91 
92  constexpr __host__ __device__ uint32_t fmix32(uint32_t h) const noexcept {
93  h ^= h >> 16;
94  h *= 0x85ebca6b;
95  h ^= h >> 13;
96  h *= 0xc2b2ae35;
97  h ^= h >> 16;
98  return h;
99  }
100  uint32_t m_seed;
101 };
